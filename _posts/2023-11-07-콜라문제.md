```python
# 빈병 a개를 주면 새 음료 b개를 준다
# n개를 가져가면 몇개를 받을까?
def solution(a, b, n) :
    answer = 0
    while (n >= a) :
        answer += (n // a) * b
        n = (n // a) * b + (b % a)
        
	return answer
```
