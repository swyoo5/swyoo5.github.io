def solution(answers) :
    answer = [0, 0, 0]
    p1 = [1, 2, 3, 4, 5]
    p2 = [2, 1, 2, 3, 2, 4, 2, 5]
    p3 = [3, 3, 1, 1, 2, 2, 4, 4, 5, 5]
    
    for i in range(len(answers)) :
        ans = answers[i]
        if (p1[i % len(p1)] == ans) :
            answer[0] += 1
		if (p2[i % len(p2)] == ans) :
            answer[1] += 1
		if (p3[i % len(p3)] == ans) :
            answer[2] += 1
            
	result = []
    for i in range(len(answer)) :
        if answer[i] == max(answer) :
            result.append(i + 1)

	return result
