내 풀이

```python
def solution(lottos, win_nums):
    answer = []
    same_ball = 0
    for my_num in lottos :
        if my_num in win_nums :
            same_ball += 1
    answer.append(min(7 - same_ball, 6))
    same_ball += lottos.count(0)
    answer.append(min(7 - same_ball, 6))
    answer.sort()
    return answer
```

def solution(lottos, win_nums) :
    rank = [6, 6, 5, 4, 3, 2, 1]

    cnt_zero = lottos.count(0)
    for my_num in lottos :
        if my_num in win_nums :
            same_ball += 1
            
	return rank[same_ball + cnt_zero], rank[same_ball]
